快速开始

几行代码即可创建一个完整的Web应用

package main

import (
    "https://github.com/zsy619/yyhertz/framework/mvc"
    "https://github.com/zsy619/yyhertz/framework/mvc/middleware"
)

type HomeController struct {
    mvc.BaseController
}

func (c *HomeController) GetIndex() {
    c.SetData("Title", "Hello World")
    c.RenderHTML("home/index.html")
}

func main() {
    app := mvc.HertzApp
    app.Use(middleware.LoggerMiddleware())
    
    homeController := &HomeController{}
    app.AutoRouters(homeController)
    
    app.Run(":8888")
}
