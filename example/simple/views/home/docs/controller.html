控制器

控制器是Hertz MVC框架的核心组件，负责处理HTTP请求和响应。

基础控制器

所有控制器都应该继承自 mvc.BaseController：

package controllers

import (
    "github.com/zsy619/yyhertz/framework/mvc"
)

type UserController struct {
    mvc.BaseController
}

// 处理 GET /user/index 请求
func (c *UserController) GetIndex() {
    c.SetData("Title", "用户列表")
    c.RenderHTML("user/index.html")
}

// 处理 POST /user/create 请求
func (c *UserController) PostCreate() {
    name := c.GetForm("name")
    email := c.GetForm("email")
    
    if name == "" || email == "" {
        c.JSON(map[string]any{
            "success": false,
            "message": "请填写完整信息",
        })
        return
    }
    
    // 保存用户逻辑...
    
    c.JSON(map[string]any{
        "success": true,
        "message": "用户创建成功",
    })
}

方法命名规则

控制器方法使用RESTful风格的命名：

HTTP方法 控制器方法 路由 说明
GET GetIndex() /controller/index 显示列表页面
GET GetShow() /controller/show 显示详情页面
POST PostCreate() /controller/create 创建新记录
PUT PutUpdate() /controller/update 更新记录
DELETE DeleteRemove() /controller/remove 删除记录

数据处理

获取请求数据

func (c *UserController) PostCreate() {
    // 获取表单数据
    name := c.GetForm("name")
    email := c.GetForm("email")
    
    // 获取URL参数
    id := c.GetParam("id")
    
    // 获取查询参数
    page := c.GetQuery("page")
    size := c.GetQuery("size")
    
    // 获取JSON数据
    var user User
    if err := c.BindJSON(&user); err != nil {
        c.JSON(map[string]any{
            "error": "JSON解析失败",
        })
        return
    }
}

设置响应数据

func (c *UserController) GetIndex() {
    users := []User{
        {ID: 1, Name: "张三", Email: "zhang@example.com"},
        {ID: 2, Name: "李四", Email: "li@example.com"},
    }
    
    // 设置模板数据
    c.SetData("Title", "用户列表")
    c.SetData("Users", users)
    c.SetData("Total", len(users))
    
    // 渲染HTML模板
    c.RenderHTML("user/index.html")
}

响应类型

JSON响应

func (c *UserController) GetAPI() {
    c.JSON(map[string]any{
        "success": true,
        "data": map[string]any{
            "users": users,
            "total": len(users),
        },
        "message": "获取成功",
    })
}

HTML响应

func (c *UserController) GetIndex() {
    c.SetData("Title", "用户列表")
    c.RenderHTML("user/index.html")
}

// 使用布局模板
func (c *UserController) GetProfile() {
    c.SetData("Title", "用户资料")
    c.RenderWithLayout("user/profile.html", "layout/layout.html")
}

文件下载

func (c *UserController) GetExport() {
    filePath := "/tmp/users.xlsx"
    c.FileDownload(filePath, "用户列表.xlsx")
}

控制器初始化

可以通过 Init 方法进行初始化操作：

type UserController struct {
    mvc.BaseController
    userService *UserService
}

func (c *UserController) Init() {
    // 初始化服务
    c.userService = NewUserService()
    
    // 设置公共数据
    c.SetData("ControllerName", "用户管理")
}

错误处理

func (c *UserController) GetShow() {
    idStr := c.GetParam("id")
    id, err := strconv.Atoi(idStr)
    if err != nil {
        c.Error("无效的用户ID")
        return
    }
    
    user, err := c.userService.GetByID(id)
    if err != nil {
        c.Error("用户不存在")
        return
    }
    
    c.SetData("User", user)
    c.RenderHTML("user/show.html")
}

完整示例

package controllers

import (
    "strconv"
    "github.com/zsy619/yyhertz/framework/mvc"
)

type BlogController struct {
    mvc.BaseController
}

// 博客列表
func (c *BlogController) GetIndex() {
    page, _ := strconv.Atoi(c.GetQuery("page"))
    if page <= 0 {
        page = 1
    }
    
    blogs := []Blog{
        {ID: 1, Title: "Hertz MVC入门", Content: "..."},
        {ID: 2, Title: "Go Web开发", Content: "..."},
    }
    
    c.SetData("Title", "博客列表")
    c.SetData("Blogs", blogs)
    c.SetData("CurrentPage", page)
    c.RenderHTML("blog/index.html")
}

// 博客详情
func (c *BlogController) GetShow() {
    id, _ := strconv.Atoi(c.GetParam("id"))
    
    blog := Blog{
        ID:      id,
        Title:   "Hertz MVC入门",
        Content: "详细内容...",
    }
    
    c.SetData("Title", blog.Title)
    c.SetData("Blog", blog)
    c.RenderHTML("blog/show.html")
}

// 创建博客
func (c *BlogController) PostCreate() {
    title := c.GetForm("title")
    content := c.GetForm("content")
    
    if title == "" || content == "" {
        c.JSON(map[string]any{
            "success": false,
            "message": "标题和内容不能为空",
        })
        return
    }
    
    // 保存博客逻辑...
    
    c.JSON(map[string]any{
        "success": true,
        "message": "博客创建成功",
        "data": map[string]any{
            "id": 123,
        },
    })
}
最佳实践
  • 控制器只负责处理HTTP请求，业务逻辑应该放在Service层
  • 使用结构体绑定来处理复杂的请求数据
  • 合理使用HTTP状态码
  • 统一错误处理和响应格式
  • 在Init方法中进行必要的初始化操作
