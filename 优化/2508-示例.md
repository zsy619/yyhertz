{{else}}
    <p>请先登录</p>
{{end}}

<!-- 复杂条件 -->
{{if and .User (eq .User.Role "admin")}}
    <a href="/admin">管理后台</a>
{{end}}

<!-- 检查是否存在 -->
{{if .Message}}
    <div class="alert alert-info">{{.Message}}</div>
{{end}}</code></pre>
